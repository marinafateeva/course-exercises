document.addEventListener('DOMContentLoaded', function() { const tabs = document.querySelectorAll('.tab'); tabs.forEach(tab => { tab.addEventListener('click', function(e) { const tabName = this.getAttribute('data-tab'); // Удаляем активный класс у всех вкладок и контента document.querySelectorAll('.tab').forEach(t => t.classList.remove('active')); document.querySelectorAll('.content').forEach(c => c.classList.remove('active')); // Добавляем активный класс кликнутой вкладке и соответствующему контенту this.classList.add('active'); document.getElementById(tabName).classList.add('active'); }); }); });
